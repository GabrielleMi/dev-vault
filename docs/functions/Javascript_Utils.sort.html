    Function sort

    Function used to sort arrays of objects.

    Example: should sort objects by key

    const test = [{ name: 'a' }, { name: 'z' }, { name: 'u' }, { name: 'b' }];
    sort(test).by('name'); // Output: [{ name: 'a' }, { name: 'b' }, { name: 'u' }, { name: 'z' }]

    Example: should sort objects by nested key

    const test = [{ author: { name: 'b' } }, { author: { name: 'z' } }, { author: { name: 'a' } }];
    sort(test).by('author.name');
    /* Output: [
      { author: { name: 'a' } },
      { author: { name: 'b' } },
      { author: { name: 'z' } }
    ] */

    Example: should use fallback key when values are the same

    const test = [{ a: 1, b: 'b' }, { a: 1, b: 'a' }, { a: 0, b: 'c' }];
    sort(test).by('a', 'b');
    /* Output: [
      { a: 0, b: 'c' },
      { a: 1, b: 'a' },
      { a: 1, b: 'b' }
    ] */

    Example: should use the function when the key is a function

    const test = [{ a: 5 }, { a: 2 }, { a: 1 }];
    sort(test).by((a, b) => `${a.a}`.localeCompare(`${b.a}`));
    /* Output: [
      { a: 1 },
      { a: 2 },
      { a: 5 }
    ] */

    Example: should sort descending

    const test = [{ a: 5 }, { a: 2 }, { a: 1 }];
    sort(test, { isDesc: true }).by('a');
    /* Output: [
      { a: 5 },
      { a: 2 },
      { a: 1 }
    ] */

    Example: should sort descending for only a specific key

    const test = [{ score: 12, name: 'John' }, { score: 12, name: 'Alicia' }, { score: 10, name: 'Joe' }];

    sort(test).by({ key: 'score', isDesc: true }, 'name'); // Output: [{ score: 12, name: 'Alicia' }, { score: 12, name: 'John' }, { score: 10, name: 'Joe' }]
